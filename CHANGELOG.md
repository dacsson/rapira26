#### Типы суммы и паттер-матчинг

```
\ Types hold variants
тип Сезон
    Зима
    Весна
    Лето
    Осень

\ Of course we can pattern match
проц сезон_в_строку(сезон)
    выбор сезон
        при Зима: возврат "Зима"
        при Весна: возврат "Весна"
        при Лето: возврат "Лето"
        при Осень: возврат "Осень"

сейчас := Лето
вывод: сезон_в_строку(сейчас)
\ => Лето

\ Variants on themselfs can hold fields
тип ШкольныйЧел
    Ученик(класс, имя)
    Учитель(имя)

функ кто_в_школе(чел)
    выбор чел
        при Ученик(класс, имя):
            возврат "в классе ученик: " + имя
        при Учитель(имя):
            возврат "в классе учитель: " + имя

чел := Ученик(11, "Тёма")
вывод: кто_в_школе(чел)
\ => в классе ученик: Тёма

\ This is how we create JUST structs
тип Точка
    Точка(х, у)

\ Structs are essentially "named-tuples"
точечка := Точка(2, 3)
вывод: "(х: ", точечка.х, ", у: ", точечка.у, ")"
\ => (х: 2, у: 3)

\ Equality of Structs
вторая := Точка(2, 3)
вывод: точечка = вторая
\ => да

\ Mutation
точечка.х := 3
вывод: "(х: ", точечка.х, ", у: ", точечка.у, ")"
\ => (х: 3, у: 3)
```

- см. `runtime-c/variant.c`
